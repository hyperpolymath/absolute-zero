To recover in a new container:

[source,bash]
----
sudo apt-get install -y chezscheme libgmp-dev
git clone --depth 1 https://github.com/idris-lang/Idris2.git /tmp/Idris2
cd /tmp/Idris2
PREFIX=$HOME/.idris2 SCHEME=scheme make bootstrap
PREFIX=$HOME/.idris2 SCHEME=scheme make install
export PATH=$HOME/.idris2/bin:$PATH
----
